## üíª Developer Guide

### Installation

```javascript
const {
  MathematicalLegacyCollection,
  MATHEMATICAL_GENIUSES,
  LEGACY_COIN_SETS
} = require('./src/nft/mathematical-legacy-collection.js');
```

### Basic Usage

#### Initialize Collection
```javascript
const collection = new MathematicalLegacyCollection({
  frequency: 963,
  enableSentientGI: true,
  enableEducationalModules: true,
  enableChronoWeaver: true
});

await collection.initialize();
```

#### Mint Legacy NFT
```javascript
const nft = await collection.mintLegacyNFT(
  'KATHERINE_JOHNSON',
  'recipient_wallet_address'
);

console.log(`Minted: ${nft.metadata.name}`);
console.log(`Token ID: ${nft.tokenId}`);
console.log(`Frequency: ${nft.attributes.frequency}Hz`);
```

#### Issue Legacy Coin
```javascript
const issuance = await collection.issueLegacyCoin(
  'PIONEER_COIN',
  'recipient_wallet_address',
  5 // amount
);

console.log(`Issued ${issuance.amount} ${issuance.coinType}`);
console.log(`Total Value: ${issuance.value}`);
```

#### Query Sentient Model
```javascript
const response = await collection.querySentient(
  'GLADYS_WEST',
  'How did your work lead to GPS technology?'
);

console.log(`Wisdom: ${response.wisdom}`);
console.log(`Teaching: ${response.teaching}`);
console.log(`Relevant Fields: ${response.relevantFields.join(', ')}`);
```

#### Access Educational Module
```javascript
const moduleAccess = await collection.accessEducationalModule(
  'Virtual Museum Tours',
  { age: 10, interests: ['space', 'computers'] }
);

console.log(`Module: ${moduleAccess.module}`);
console.log(`Age Appropriate: ${moduleAccess.ageAppropriate}`);
```

### Query Methods

#### Get All Genius Profiles
```javascript
const profiles = collection.getAllGeniusProfiles();
profiles.forEach(profile => {
  console.log(`${profile.name} - ${profile.title}`);
  console.log(`Frequency: ${profile.frequency}Hz`);
  console.log(`Rarity: ${profile.rarityTier}`);
});
```

#### Get Coin Holdings
```javascript
const holdings = collection.getCoinHoldings('wallet_address');
holdings.forEach(holding => {
  console.log(`${holding.name} (${holding.symbol})`);
  console.log(`Frequency: ${holding.frequency}Hz`);
});
```

#### Get Statistics
```javascript
const stats = collection.getStatistics();
console.log(`Total Legacy NFTs: ${stats.totalLegacyNFTs}`);
console.log(`Active Sentients: ${stats.activeSentients}`);
console.log(`Educational Access: ${stats.educationalAccess}`);
console.log(`Coins Issued: ${stats.coinsIssued}`);
console.log(`Eternal Memory Nodes: ${stats.eternalmemoryNodes}`);
console.log(`N-GI Core Connected: ${stats.ngiCoreConnected}`);
```
