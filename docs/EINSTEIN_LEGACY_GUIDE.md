## Getting Started

### Installation

```bash
npm install
```

### Basic Usage

```javascript
const EducationalSystem = require('./src/education');
const EinsteinAICompanion = require('./src/education/einstein-ai-companion');
const EinsteinKnowledgeQuest = require('./src/education/einstein-knowledge-quest');

// Initialize systems
const educationalSystem = new EducationalSystem({ frequency: 528 });
const companion = new EinsteinAICompanion({ frequency: 528 });
const questSystem = new EinsteinKnowledgeQuest({ frequency: 528 });

await educationalSystem.initialize();
await companion.initialize();
await questSystem.initialize();

// Start complete learning experience
const experience = await educationalSystem.startEducationalExperience(
  'einstein_legacy',
  'student_001',
  {
    childName: 'Young Scholar',
    childAge: 14,
    initialBalance: 1000
  }
);

// Interact with Einstein AI
const session = companion.startSession('student_001', {
  topic: 'relativity',
  knowledgeLevel: 'beginner'
});

const response = companion.interact(session.session.sessionId, 
  'How does time dilation work?'
);

// Start quest
const quest = questSystem.startQuest('student_001', 'relativity_1');

// Complete tasks
questSystem.completeTask('student_001', 'relativity_1', 0);
questSystem.completeTask('student_001', 'relativity_1', 1);
questSystem.completeTask('student_001', 'relativity_1', 2);
const completion = questSystem.completeTask('student_001', 'relativity_1', 3);

console.log(`Quest completed! Earned ${completion.rewards.einsteinCoins} EINSTEIN coins`);
```
